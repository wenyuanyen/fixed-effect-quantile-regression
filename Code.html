1. Model setup¶

Generate the one-way fixed effect model

$$Y_{ij}=X_{ij}\beta+ \alpha_i + \varepsilon_{ij}~~~~~\text{where}~~~X_{ij} \in \mathbb{R} ~~~~\beta \in \mathbb{R}$$$$i=1,2,...,n~~~~~~~j=1,2,...,m_i~~~~~~~N:=\sum_{i=1}^n m_i$$

Denote $\boldsymbol{Y_i}$ $(m_i \times 1)$ the vector consisting of $m_i$ observed outcomes of $i$-th unit, $i=1,2,...,n$, and $\boldsymbol{X_i}$ the vector consisting of $m_i$ observed covariate of $i$-th unit. So, $\boldsymbol{Y_i} = (Y_{i1},Y_{i2},...,Y_{i m_i})'$, $\boldsymbol{X_i} = (X_{i1},X_{i2},...,X_{i m_i})'$. The errors are $\boldsymbol{\varepsilon_i} = (\varepsilon_{i1},\varepsilon_{i2},...,\varepsilon_{i m_i})'$. Denote $\boldsymbol{1}_{m_i}$ the $(m_i \times 1)$ vector of all ones. Then we can present the linear model as follows

$$\underbrace{\begin{bmatrix} \boldsymbol{Y_1} \\ \boldsymbol{Y_2} \\ \vdots \\ \boldsymbol{Y_{n}} \\ \end{bmatrix}}_{\boldsymbol{Y}} = \underbrace{\begin{bmatrix} \boldsymbol{X_1} \\ \boldsymbol{X_2} \\ \vdots \\ \boldsymbol{X_{n}} \\ \end{bmatrix}}_{\boldsymbol{X}} \beta + \underbrace{\begin{bmatrix} \boldsymbol{1}_{m_1} & \boldsymbol{0} & \boldsymbol{0} & \dots & \boldsymbol{0} \\ \boldsymbol{0} & \boldsymbol{1}_{m_2} & \boldsymbol{0} & \dots & \boldsymbol{0} \\ \vdots & \vdots & \vdots & \ddots & \vdots \\ \boldsymbol{0} & \boldsymbol{0} & \boldsymbol{0} & \dots & \boldsymbol{1}_{m_n} \\ \end{bmatrix}}_{\boldsymbol{Z}} \underbrace{\begin{bmatrix} \alpha_1 \\ \alpha_2 \\ \vdots \\ \alpha_n \\ \end{bmatrix}}_{\boldsymbol{\alpha}} + \underbrace{\begin{bmatrix} \boldsymbol{\varepsilon_{1}} \\ \boldsymbol{\varepsilon_2} \\ \vdots \\ \boldsymbol{\varepsilon_{n}} \\ \end{bmatrix}}_{\boldsymbol{\varepsilon}} \iff \boldsymbol{Y}=\boldsymbol{X}\beta+\boldsymbol{Z}\boldsymbol{\alpha}+\boldsymbol{\varepsilon}$$

def Z_generator(n, M):
    I = np.diag(np.ones(n))
    Z = I.copy()
    for k in range(M-1):
        Z = np.concatenate((Z,I), axis=0)
    return(Z)

def standardize(X, Y, tau):
    n, p = X.shape
    # transform into standard form
    c = np.concatenate((np.zeros(2*p), tau * np.ones(n), (1-tau) * np.ones(n)), axis=None)
    A = np.concatenate((X, -X, np.diag(np.ones(n)), np.diag(-np.ones(n))), axis=1)
    b = Y
    return(A, b, c)

(F3-1) Gurobi optimization on primal QR standard form¶

\begin{aligned} &\underset{\theta,u,v}{\text{min}}&& \tau 1_{N} u + (1-\tau) 1_{N} v\\ &\text{subject to} && \begin{bmatrix} \boldsymbol{\tilde X} & - \boldsymbol{\tilde X} & \boldsymbol{I} & -\boldsymbol{I} \end{bmatrix} \begin{bmatrix}\beta_+ \\ \beta_- \\ \boldsymbol{\alpha}_+ \\ \boldsymbol{\alpha}_- \\u \\v \end{bmatrix}=Y\\ &&& \beta_+, ~\beta_- ,~\boldsymbol{\alpha}_+, ~\boldsymbol{\alpha}_-,~u ,~v \ge 0 \\ &&& \text{where } \boldsymbol{\tilde X} = \begin{bmatrix} \boldsymbol{X} & \boldsymbol{Z}\end{bmatrix} \end{aligned}

def Gurobi_qreg(Y, X, Z, tau = 0.5): 
    design_matrix  = np.concatenate((X, Z), axis=1)
    m, p           = design_matrix.shape # m is assigned N; p is assigned (n+p)
    A, b, c        = standardize(design_matrix, Y, tau = tau)
    model      = Model()
    rows, cols = len(b), len(c)    # same b, c 
    for j in range(cols):
        model.addVar(lb = 0, obj = c[j])  # lower bounds are 0 <= x
    model.update()

    var = model.getVars()
    S   = coo_matrix(A)       # same A matrix
    expr, used = [], []
    for i in range(rows):
        expr.append(LinExpr())
        used.append(False)
    for i, j, s in zip(S.row, S.col, S.data):
        expr[i] += s*var[j]
        used[i] = True
    for i in range(rows):
        if used[i]:
            model.addConstr(lhs=expr[i], sense=GRB.EQUAL, rhs=b[i])
    model.update()
    model.ModelSense = 1  # minimization
    model.setParam('OutputFlag', 0) # mute the output
    model.optimize()
    
    solution = np.array(model.getAttr ('X', var))
    beta_gurobi  = solution[:p]           -  solution[p:(2*p)]       # beta+ - beta-
    error_gurobi = solution[(2*p):(2*p+m)]-  solution[(2*p+m):]      # u - v
    
    time_lapse = model.getAttr ('Runtime')
    
    return(beta_gurobi, error_gurobi, solution, time_lapse)

(F3-2) Gurobi optimization on dual QR standard form¶

\begin{aligned} &\underset{a}{\text{max}}&& y^{\top}a \\ &\text{subject to} && \boldsymbol{\tilde X}^{\top}a = (1-\tau)\boldsymbol{\tilde X}^{\top}\boldsymbol{1}_N\\ &&& a \in [0,1]^N ~~~N=\sum_{i=1}^n m_i \\ \end{aligned}

def Gurobi_qreg_dual(Y, X, n, M, N, tau = 0.5): 
    model      = Model()
    for k in range(N):
        model.addVar(lb = 0, ub = 1, obj = Y[k])  #  0 <= a <= 1
    model.update()
    var = model.getVars()
    
    b_X = np.sum(X) * (1-tau)
    expr_X = LinExpr()
    for k in range(N):
        expr_X += X[k] * var[k]
    model.addConstr(lhs = expr_X, sense=GRB.EQUAL, rhs= b_X )
    
    b_Z = M * (1-tau)
    expr_Z = []
    for i in range(n):
        expr_Z.append(LinExpr())
    for i in range(n):
        for k in range(i, N, n):
            expr_Z[i] +=   var[k]
    for i in range(n):
        model.addConstr(lhs=expr_Z[i], sense=GRB.EQUAL, rhs=b_Z)

    model.update()
    model.ModelSense = -1  # maximization
    model.setParam('OutputFlag', 0) # mute the output
    model.optimize()
    
    solution = np.array(model.getAttr ('X', var))
    c = np.where((solution == 0) + (solution == 1) == False) # indices of support vectors
    c = np.array(c)[0]
    X_tilde = np.zeros((n+1,n+1))
    X_tilde[:,0] = X[c]
    for l in range(n+1):
        group = np.mod(c[l],n)
        X_tilde[l, group+1] = 1
    X_tilde      =  scipy.sparse.csr_matrix(X_tilde)
    beta_gurobi  = scipy.sparse.linalg.spsolve(X_tilde, Y[c].T)    
    time_lapse   = model.getAttr ('Runtime')
    
    return(beta_gurobi, c, solution, time_lapse)

def check_support(a,n,M,N):
    check = []
    add   = []
    for i in range(n):
        current = a[range(i, N, n)]
        ind_01  = (current == 0 ) + (current == 1 )
        if False in ind_01:
            check.append(True)
        if sum(ind_01) <  M-1:
            add.append(i) 
    return(sum(check), add[0])

3. Simulate the dataset¶

Mean equations¶

Location shift: $$Y_{ij}= \alpha_i + X_{ij} + \varepsilon_{ij} ~~~\text{where }X_{ij}=0.3 \alpha_i +w_{ij}$$ Location-scale shift:$$Y_{ij}= \alpha_i + X_{ij} + (1+\gamma X_{ij})\varepsilon_{ij} ~~~\text{where }X_{ij}=0.3 \alpha_i +w_{ij}$$

Distributional assumptions¶

$$w_{ij}\stackrel{\text{iid}}{\sim}\chi^2_3~~~~\varepsilon_{ij} \stackrel{\text{iid}}{\sim} F=N(0,1)~~~~\alpha_{i} \stackrel{\text{iid}}{\sim} \text{Uniform}[0,1]$$

ns = [25,50,100,200,500,1000,2500,5000,7500,10000]
Ms = [5,10,15,20,25,35,40,45,50,100]
rows = len(ns)
cols = len(Ms)
time1 = np.zeros((rows, cols))
beta1 = np.zeros((rows, cols))
alpha1= np.zeros((rows, cols))
SV1_a   = np.zeros((rows, cols))
SV1_b   = np.zeros((rows, cols))

time2 = np.zeros((rows, cols))
beta2 = np.zeros((rows, cols))
alpha2= np.zeros((rows, cols))
SV2_a   = np.zeros((rows, cols))
SV2_b   = np.zeros((rows, cols))


tau  = 0.50001

for i in range(rows):
    for j in range(cols):
        
        """Generate FE, covariates and epsilon"""
        n, M  = ns[i], Ms[j]
        gamma = 1
        N     = n * M

        u     = np.random.chisquare(3, size = N)
        alpha = np.random.uniform(size = n)
        eps   = np.random.normal(size = N)

        X   = np.zeros(N)
        for k in range(N):
            fe_i = np.mod(k, n)
            X[k] = 0.3 * alpha[fe_i] + u[k]
            
        """Location-scale shift"""
        
        Y_locsc = np.tile(alpha, M) + X + (1 + gamma * X) * eps
        beta_locsc, sv_locsc, a_locsc, time_locsc = Gurobi_qreg_dual(Y_locsc, X, n, M, N, tau = tau)
        
        # check support vectors
        SV1_a[i,j], SV1_b[i,j] = check_support(a_locsc, n, M, N)
        
        # appedn fitted beta, alpha bias, and run time
        beta1[i,j]  = beta_locsc[0]
        alpha1[i,j] = np.max(np.abs(beta_locsc[1:] - alpha))
        time1[i,j]  = time_locsc
        
        
        """Location shift"""
        
        Y_loc   = np.tile(alpha, M) + X +  eps
        beta_loc  , sv_loc  , a_loc  , time_loc   = Gurobi_qreg_dual(Y_loc  , X, n, M, N, tau = tau)
        
        # check support vectors
        SV2_a[i,j], SV2_b[i,j] = check_support(a_loc, n, M, N)
        
        
        # appedn fitted beta, alpha bias, and run time
        beta2[i,j]  = beta_loc[0]
        alpha2[i,j] = np.max(np.abs(beta_loc[1:] - alpha))
        time2[i,j]  = time_loc
